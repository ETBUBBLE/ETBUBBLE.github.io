  汇编语言，写一个简单加法程序“3+5=8”，加数和结果分别存在内存单元NUM1、NUM2、NUM3中。

 
```
 data segment
  num1 db 3
  num2 db 5
  num3 db ?
data ends
code segment
assume cs:code,ds:data
 start:mov ax,data
    mov ds,ax
    mov al,num1
    add al,num2
    mov num3,al
    mov ax,4c00h
    int 21h
code ends
end start

```
